## Overview
Build an admin panel where staff can manage users, view all projects/agreements, configure system settings (Puppeteer URL), and handle project action steps that push updates to user-facing dashboards.

## 2. Admin Middleware

```javascript
// /middleware/adminAuth.js

const db = require('../db');

const authenticateAdmin = async (req, res, next) => {
  try {
    // First check if user is authenticated
    if (!req.user || !req.user.id) {
      return res.status(401).json({ error: 'Authentication required' });
    }
    
    // Check if user has admin role
    const result = await db.query(
      'SELECT role, email, full_name FROM users WHERE id = $1',
      [req.user.id]
    );
    
    if (result.rows.length === 0) {
      return res.status(403).json({ error: 'Access denied' });
    }
    
    const user = result.rows[0];
    
    if (!['admin', 'super_admin', 'staff'].includes(user.role)) {
      return res.status(403).json({ error: 'Admin access required' });
    }
    
    // Add role and full user info to request
    req.adminUser = {
      id: req.user.id,
      email: user.email,
      full_name: user.full_name,
      role: user.role
    };
    
    next();
    
  } catch (error) {
    console.error('Admin auth error:', error);
    res.status(500).json({ error: 'Authentication failed' });
  }
};

// Check for super admin (higher privileges)
const authenticateSuperAdmin = async (req, res, next) => {
  try {
    await authenticateAdmin(req, res, () => {
      if (req.adminUser.role !== 'super_admin') {
        return res.status(403).json({ error: 'Super admin access required' });
      }
      next();
    });
  } catch (error) {
    console.error('Super admin auth error:', error);
    res.status(500).json({ error: 'Authentication failed' });
  }
};

module.exports = { authenticateAdmin, authenticateSuperAdmin };
```

```javascript
const express = require('express');
const router = express.Router();
const { authenticateUser } = require('../../middleware/auth');
const { authenticateAdmin } = require('../../middleware/adminAuth');
const db = require('../../db');

// All routes require authentication
router.use(authenticateUser);
router.use(authenticateAdmin);

// GET all users
router.get('/', async (req, res) => {
  try {
    const { role, search } = req.query;
    
    let query = `
      SELECT 
        u.id,
        u.email,
        u.full_name,
        u.company_name,
        u.phone,
        u.role,
        u.created_at,
        u.last_login_at,
        u.is_active,
        COUNT(DISTINCT p.id) as project_count,
        COUNT(DISTINCT d.id) as agreement_count
      FROM users u
      LEFT JOIN projects p ON u.id = p.user_id
      LEFT JOIN documents d ON u.id = d.user_id
      WHERE 1=1
    `;
    
    const params = [];
    
    if (role) {
      params.push(role);
      query += ` AND u.role = $${params.length}`;
    }
    
    if (search) {
      params.push(`%${search}%`);
      query += ` AND (u.email ILIKE $${params.length} OR u.full_name ILIKE $${params.length})`;
    }
    
    query += `
      GROUP BY u.id
      ORDER BY u.created_at DESC
    `;
    
    const result = await db.query(query, params);
    
    res.json({ users: result.rows });
    
  } catch (error) {
    console.error('Get users error:', error);
    res.status(500).json({ error: 'Failed to get users' });
  }
});

      SELECT 
        id, title, status, created_at, sent_at, completed_at
```

```javascript
// /routes/admin/projects.js

const express = require('express');
const router = express.Router();
const { authenticateUser } = require('../../middleware/auth');
const { authenticateAdmin } = require('../../middleware/adminAuth');
const db = require('../../db');

router.use(authenticateUser);
router.use(authenticateAdmin);

// GET all projects (across all users)
router.get('/', async (req, res) => {
  try {
    const { status, stage, user_id } = req.query;
    
    let query = `
      SELECT 
        p.*,
        u.email as user_email,
        u.full_name as user_name,
        u.company_name as user_company,
        COUNT(DISTINCT pt.id) FILTER (WHERE pt.status = 'completed') as completed_tasks,
        COUNT(DISTINCT pt.id) as total_tasks,
        COUNT(DISTINCT at.id) FILTER (WHERE at.status = 'pending') as pending_admin_tasks
      FROM projects p
      LEFT JOIN users u ON p.user_id = u.id
      LEFT JOIN project_tasks pt ON p.id = pt.project_id
      LEFT JOIN admin_tasks at ON p.id = at.project_id
      WHERE 1=1
    `;
    
    const params = [];
    
    if (status) {
      params.push(status);
      query += ` AND p.status = $${params.length}`;
    }
    
    if (stage) {
      params.push(stage);
      query += ` AND p.current_stage = $${params.length}`;
    }
    
    if (user_id) {
      params.push(user_id);
      query += ` AND p.user_id = $${params.length}`;
    }
    
    query += `
      GROUP BY p.id, u.email, u.full_name, u.company_name
      ORDER BY p.last_updated DESC
    `;
    
    const result = await db.query(query, params);
    
    res.json({ projects: result.rows });
    
  } catch (error) {
    console.error('Get admin projects error:', error);
    res.status(500).json({ error: 'Failed to get projects' });
  }
});

// GET single project (admin view with admin tasks)
router.get('/:id', async (req, res) => {
  try {
    // Get project details
    const projectResult = await db.query(`
      SELECT 
        p.*,
        u.email as user_email,
        u.full_name as user_name,
        u.company_name as user_company,
        d.title as agreement_title,
        d.file_path as agreement_file
      FROM projects p
      LEFT JOIN users u ON p.user_id = u.id
      LEFT JOIN documents d ON p.agreement_id = d.id
      WHERE p.id = $1
    `, [req.params.id]);
    
    if (projectResult.rows.length === 0) {
      return res.status(404).json({ error: 'Project not found' });
    }
    
    // Get admin tasks (internal workflow)
    const adminTasksResult = await db.query(`
      SELECT 
        at.*,
        assigned_user.full_name as assigned_to_name,
        completed_user.full_name as completed_by_name,
        ps.stage_name as milestone_stage_name,
        pt.task_title as milestone_task_title
      FROM admin_tasks at
      LEFT JOIN users assigned_user ON at.assigned_to = assigned_user.id
      LEFT JOIN users completed_user ON at.completed_by = completed_user.id
      LEFT JOIN project_stages ps ON at.user_milestone_stage_id = ps.id
      LEFT JOIN project_tasks pt ON at.user_milestone_task_id = pt.id
      WHERE at.project_id = $1
      ORDER BY 
        CASE at.status
          WHEN 'pending' THEN 1
          WHEN 'in_progress' THEN 2
          WHEN 'completed' THEN 3
          WHEN 'blocked' THEN 4
        END,
        CASE at.priority
          WHEN 'critical' THEN 1
          WHEN 'high' THEN 2
          WHEN 'medium' THEN 3
          WHEN 'low' THEN 4
        END,
        at.created_at
    `, [req.params.id]);
    
    // Get user-facing stages/tasks
    const userStagesResult = await db.query(`
      SELECT 
        ps.*,
        COALESCE(
          json_agg(
            json_build_object(
              'id', pt.id,
              'task_title', pt.task_title,
              'status', pt.status,
              'completed_at', pt.completed_at
            ) ORDER BY pt.id
          ) FILTER (WHERE pt.id IS NOT NULL), '[]'
        ) as tasks
      FROM project_stages ps
      LEFT JOIN project_tasks pt ON ps.id = pt.stage_id
      WHERE ps.project_id = $1
      GROUP BY ps.id
      ORDER BY ps.stage_order
    `, [req.params.id]);
    
    // Get admin activity
    const adminActivityResult = await db.query(`
      SELECT 
        aa.*,
        u.full_name as user_name
      FROM admin_activity aa
      LEFT JOIN users u ON aa.user_id = u.id
      WHERE aa.project_id = $1
      ORDER BY aa.created_at DESC
      LIMIT 100
    `, [req.params.id]);
    
    // Get documents
    const documentsResult = await db.query(`
      SELECT *
      FROM project_documents
      WHERE project_id = $1
      ORDER BY uploaded_at DESC
    `, [req.params.id]);
    
    res.json({
      project: projectResult.rows[0],
      admin_tasks: adminTasksResult.rows,
      user_stages: userStagesResult.rows,
      admin_activity: adminActivityResult.rows,
      documents: documentsResult.rows
    });
    
  } catch (error) {
    console.error('Get admin project error:', error);
    res.status(500).json({ error: 'Failed to get project details' });
  }
});

// UPDATE admin task (complete, assign, etc.)
router.patch('/:projectId/admin-tasks/:taskId', async (req, res) => {
  try {
    const {
      status,
      assigned_to,
      internal_notes,
      document_id
    } = req.body;
    
    const updates = [];
    const values = [];
    let paramIndex = 1;
    
    if (status !== undefined) {
      updates.push(`status = $${paramIndex}`);
      values.push(status);
      paramIndex++;
      
      if (status === 'completed') {
        updates.push(`completed_at = NOW()`);
        updates.push(`completed_by = $${paramIndex}`);
        values.push(req.adminUser.id);
        paramIndex++;
      }
    }
    
    if (assigned_to !== undefined) {
      updates.push(`assigned_to = $${paramIndex}`);
      values.push(assigned_to);
      paramIndex++;
    }
    
    if (internal_notes !== undefined) {
      updates.push(`internal_notes = $${paramIndex}`);
      values.push(internal_notes);
      paramIndex++;
    }
    
    if (document_id !== undefined) {
      updates.push(`document_id = $${paramIndex}`);
      values.push(document_id);
      paramIndex++;
    }
    
    if (updates.length === 0) {
      return res.status(400).json({ error: 'No valid fields to update' });
    }
    
    values.push(req.params.taskId);
    
    const result = await db.query(`
      UPDATE admin_tasks
      SET ${updates.join(', ')}
      WHERE id = $${paramIndex} AND project_id = $${paramIndex + 1}
      RETURNING *
    `, [...values, req.params.projectId]);
    
    if (result.rows.length === 0) {
      return res.status(404).json({ error: 'Task not found' });
    }
    
    const task = result.rows[0];
    
    // If task completed and auto_update enabled, update user-facing task
    if (status === 'completed' && task.auto_update_user_task && task.user_milestone_task_id) {
      await db.query(`
        UPDATE project_tasks
        SET status = 'completed', completed_at = NOW(), completed_by = 'Admin'
        WHERE id = $1
      `, [task.user_milestone_task_id]);
      
      // Log to user-visible activity
      await db.query(`
        INSERT INTO project_activity (
          project_id,
          activity_type,
          activity_description,
          visible_to_borrower
        ) VALUES ($1, 'task_completed', $2, true)
      `, [
        req.params.projectId,
        `Task "${task.task_title}" completed`
      ]);
      
      // Recalculate project progress
      await updateProjectProgress(req.params.projectId);
    }
    
    // Log admin activity
    await db.query(`
      INSERT INTO admin_activity (project_id, user_id, action_type, action_description)
      VALUES ($1, $2, 'task_updated', $3)
    `, [
      req.params.projectId,
      req.adminUser.id,
      `Admin task "${task.task_title}" updated to ${status}`
    ]);
    
    res.json({ task: result.rows[0] });
    
  } catch (error) {
    console.error('Update admin task error:', error);
    res.status(500).json({ error: 'Failed to update task' });
  }
});

// Helper function to update project progress
const updateProjectProgress = async (projectId) => {
  const result = await db.query(`
    SELECT 
      COUNT(*) FILTER (WHERE status = 'completed') as completed,
      COUNT(*) as total
    FROM project_tasks
    WHERE project_id = $1
  `, [projectId]);
  
  const { completed, total } = result.rows[0];
  const percentage = total > 0 ? Math.round((completed / total) * 100) : 0;
  
  await db.query(
    'UPDATE projects SET progress_percentage = $1, last_updated = NOW() WHERE id = $2',
    [percentage, projectId]
  );
};
```

```javascript
      SELECT 
        ps.id as stage_id,
        ps.stage_order,
        json_agg(
          json_build_object(
            'id', pt.id,
            'task_title', pt.task_title
          )
        ) as tasks
      FROM project_stages ps
      LEFT JOIN project_tasks pt ON ps.id = pt.stage_id
      WHERE ps.project_id = $1
      GROUP BY ps.id, ps.stage_order
      ORDER BY ps.stage_order

        stage_id: stage.stage_id,
        tasks: stage.tasks
      };

    // Create admin tasks from template
    for (const template of ADMIN_TASK_TEMPLATES) {
      const stageData = stagesMap[template.user_milestone_stage_order];
      
      if (!stageData) continue;
      
      // Find matching user milestone task
      const userTask = stageData.tasks.find(t => 
        t.task_title === template.user_milestone_task_title
      );

          project_id,
          task_title,
          task_description,
          task_category,
          priority,
          user_milestone_stage_id,
          user_milestone_task_id,
          auto_update_user_task

        template.task_description,
```

```jsx
import { useState, useEffect } from 'react';

const AdminSettings = () => {
  const [settings, setSettings] = useState([]);

  }, []);
  
  const fetchSettings = async () => {
    try {
      const response = await fetch('/api/admin/settings', {
        credentials: 'include'
      });
      
      const data = await response.json();
      setSettings(data.settings);
    } catch (error) {
      console.error('Failed to fetch settings:', error);
    } finally {
      setLoading(false);
    }
  };
  
  const handleEdit = (setting) => {
    setEditing(setting.setting_key);
    setEditValue(setting.setting_value);
  };
  
  const handleSave = async (key) => {
    try {
      await fetch(`/api/admin/settings/${key}`, {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        credentials: 'include',
        body: JSON.stringify({ setting_value: editValue })
      });
      
      setEditing(null);
      fetchSettings();
    } catch (error) {
      console.error('Failed to update setting:', error);
      alert('Failed to update setting');
    }
  };
  
  const handleCancel = () => {
    setEditing(null);
    setEditValue('');
  };
  
  if (loading) return <div>Loading settings...</div>;
  
  return (
    <div className="admin-settings-page">
      <header>
        <h1>System Settings</h1>
        <p>Configure system-wide settings and integrations</p>

      <div className="settings-list">

                <p className="setting-description">{setting.setting_description}</p>

                    <input
                      type="url"
                      value={editValue}
                      onChange={(e) => setEditValue(e.target.value)}
                      className="setting-input"
                      placeholder="https://example.com"
                    />
                  ) : (
                    <textarea
                      value={editValue}
                      onChange={(e) => setEditValue(e.target.value)}
                      className="setting-textarea"
                      rows={3}
                    />
                  )}
                  
                  <div className="edit-actions">
                    <button onClick={() => handleSave(setting.setting_key)} className="btn-save">
                      Save
                    </button>

                      Cancel
                    </button>
                  </div>
                </div>
              ) : (
                <div className="view-mode">
                  <div className="setting-value">
                    {setting.setting_key === 'puppeteer_quote_url' ? (
```

## Complete Implementation Summary

**What This Gives You:**

1. **User Management**: View all users, edit details, deactivate accounts, see their projects/agreements
2. **Project Workflow**: Admin-side tasks that automatically update user-facing milestones when completed
3. **System Settings**: Change Puppeteer URL and other configs from the UI
4. **Full Visibility**: See all projects, all agreements, all activity across all users
5. **Two-Sided System**: Admin completes internal tasks → User sees progress on their dashboard
6. **Activity Tracking**: Separate admin activity log + user-visible activity feed

**How It Works:**
- User creates/receives project → Admin tasks auto-created
- Admin completes tasks (review docs, order appraisal, etc.)
- When admin task completed → Linked user milestone task auto-completes
- User sees updated progress bar and milestone completion
- Admin can add documents, notes, override stages
- All changes logged and tracked

**Security:**
- Role-based access (admin, staff, super_admin)
- Admin middleware on all admin routes
- Regular users cannot access admin features
- Super admin required for critical settings

This is production-ready admin panel that doesn't affect any existing user-facing functionality!
