Reference implementation (tested and verified in the pricer console):
javascriptfunction extractOptionData(li) {
  const fiberKey = Object.keys(li).find(k => k.startsWith('__reactFiber$'));
  if (!fiberKey) return null;
  let f = li[fiberKey];
  for (let d = 0; d < 6 && f; d++) f = f.return;
  const opt = f?.memoizedProps?.option;
  if (!opt || typeof opt.value !== 'string') return null;
  // Validate it looks like a MongoDB ObjectId
  if (!/^[a-f0-9]{24}$/i.test(opt.value)) return null;
  return { id: opt.value, label: opt.text, order: opt.order_num };
}
