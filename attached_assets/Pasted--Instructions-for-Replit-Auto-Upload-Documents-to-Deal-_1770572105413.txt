```javascript
// When creating a new deal and Google Drive folder
async function createDealWithDriveFolder(dealData) {
  // 1. Create deal in database
  const deal = await db.insert('deals', {
    ...dealData,
    created_at: new Date()
  });
  
  const dealId = deal.id;
  
  // 2. Create Google Drive folder
  const parentFolderId = '1e4BT2y1TTtnizDwrZl-V5jpeKthCWgYG'; // Your parent folder
  
  const driveFolder = await createDriveFolder({
    name: `Deal ${dealId} - ${dealData.borrower_name || dealData.property_address}`,
    parentId: parentFolderId,
    accessToken: dealData.user_access_token // From logged-in user's session
  });
  
  // 3. CRITICAL: Store the Drive folder ID in the deal record
  await db.query(`
    UPDATE deals 
    SET drive_folder_id = ?, drive_folder_url = ?
    WHERE id = ?
  `, [
    driveFolder.id,
    `https://drive.google.com/drive/folders/${driveFolder.id}`,
    dealId
  ]);
  
  console.log(`Created Drive folder ${driveFolder.id} for deal ${dealId}`);
  
  return { ...deal, drive_folder_id: driveFolder.id };
}

// Helper function to create Drive folder
async function createDriveFolder({ name, parentId, accessToken }) {
  const response = await fetch('https://www.googleapis.com/drive/v3/files', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${accessToken}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      name: name,
      mimeType: 'application/vnd.google-apps.folder',
      parents: [parentId]
    })
  });
  
  if (!response.ok) {
    throw new Error(`Failed to create Drive folder: ${await response.text()}`);
  }
  
  return await response.json();
}
```

## STEP 2: Upload Documents to Drive Folder on Document Upload

**Modify the document upload handler:**

```javascript
// Route: POST /api/deals/:dealId/documents/upload
async function handleDocumentUpload(req, res) {
  const { dealId } = req.params;
  const file = req.file; // Assuming you're using multer or similar
  const { document_id } = req.body; // The deal_documents record ID
  
  try {
    // 1. Get deal's Drive folder ID
    const deal = await db.query(`
      SELECT drive_folder_id, drive_folder_url 
      FROM deals 
      WHERE id = ?
    `, [dealId]);
    
    if (!deal || !deal.drive_folder_id) {
      return res.status(400).json({ 
        error: 'Deal does not have a Google Drive folder. Please contact support.' 
      });
    }
    
    // 2. Get user's access token from session
    const accessToken = req.session.google_access_token;
    
    if (!accessToken) {
      return res.status(401).json({ 
        error: 'Not authenticated with Google. Please log in again.' 
      });
    }
    
    // 3. Upload file to Google Drive
    const driveFile = await uploadFileToDrive({
      file: file,
      fileName: file.originalname,
      folderId: deal.drive_folder_id,
      accessToken: accessToken
    });
    
    console.log(`Uploaded file ${driveFile.id} to Drive folder ${deal.drive_folder_id}`);
    
    // 4. Update document record with Drive info
    await db.query(`
      UPDATE deal_documents 
      SET 
        drive_file_id = ?,
        drive_file_url = ?,
        uploaded_file = ?,
        uploaded_at = ?,
        status = 'completed'
      WHERE id = ?
    `, [
      driveFile.id,
      driveFile.webViewLink,
      file.originalname,
      new Date(),
      document_id
    ]);
    
    res.json({
      success: true,
      file: {
        id: driveFile.id,
        name: file.originalname,
        url: driveFile.webViewLink
      }
    });
    
  } catch (error) {
    console.error('Document upload error:', error);
    res.status(500).json({ error: error.message });
  }
}

// Helper function to upload file to Drive
async function uploadFileToDrive({ file, fileName, folderId, accessToken }) {
  // Create form data with file metadata
  const metadata = {
    name: fileName,
    parents: [folderId]
  };
  
  const form = new FormData();
  form.append('metadata', new Blob([JSON.stringify(metadata)], { type: 'application/json' }));
  form.append('file', file.buffer, { filename: fileName });
  
  const response = await fetch('https://www.googleapis.com/upload/drive/v3/files?uploadType=multipart&fields=id,name,webViewLink,webContentLink', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${accessToken}`
    },
    body: form
  });
  
  if (!response.ok) {
    const error = await response.text();
    throw new Error(`Drive upload failed: ${error}`);
  }
  
  return await response.json();
}
```

**Alternative: If using file.buffer approach doesn't work, try this:**

```javascript
async function uploadFileToDrive({ file, fileName, folderId, accessToken }) {
  // Step 1: Create file metadata
  const metadataResponse = await fetch('https://www.googleapis.com/drive/v3/files', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${accessToken}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      name: fileName,
      parents: [folderId]
    })
  });
  
  if (!metadataResponse.ok) {
    throw new Error(`Failed to create file metadata: ${await metadataResponse.text()}`);
  }
  
  const fileMetadata = await metadataResponse.json();
  
  // Step 2: Upload file content
  const uploadResponse = await fetch(`https://www.googleapis.com/upload/drive/v3/files/${fileMetadata.id}?uploadType=media`, {
    method: 'PATCH',
    headers: {
      'Authorization': `Bearer ${accessToken}`,
      'Content-Type': file.mimetype || 'application/octet-stream'
    },
    body: file.buffer
  });
  
  if (!uploadResponse.ok) {
    throw new Error(`Failed to upload file content: ${await uploadResponse.text()}`);
  }
  
  // Step 3: Get file details with webViewLink
  const detailsResponse = await fetch(`https://www.googleapis.com/drive/v3/files/${fileMetadata.id}?fields=id,name,webViewLink,webContentLink`, {
    headers: {
      'Authorization': `Bearer ${accessToken}`
    }
  });
  
  return await detailsResponse.json();
}
```

## STEP 5: Handle OAuth Token Refresh

**Important: Access tokens expire after 1 hour. Add refresh logic:**

```javascript
async function getValidAccessToken(req) {
  const accessToken = req.session.google_access_token;
  const refreshToken = req.session.google_refresh_token;
  const expiresAt = req.session.token_expires_at;
  
  // Check if token is still valid
  if (accessToken && Date.now() < expiresAt) {
    return accessToken;
  }
  
  // Token expired, refresh it
  if (refreshToken) {
    const response = await fetch('https://oauth2.googleapis.com/token', {
      method: 'POST',
      headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
      body: new URLSearchParams({
        client_id: process.env.GOOGLE_CLIENT_ID,
        client_secret: process.env.GOOGLE_CLIENT_SECRET,
        refresh_token: refreshToken,
        grant_type: 'refresh_token'
      })
    });
    
    const tokens = await response.json();
    
    // Update session
    req.session.google_access_token = tokens.access_token;
    req.session.token_expires_at = Date.now() + (tokens.expires_in * 1000);
    
    return tokens.access_token;
  }
  
  throw new Error('No valid Google token available. Please log in again.');
}

// Use in upload function:
const accessToken = await getValidAccessToken(req);
```

## STEP 6: Error Handling and User Feedback

**Add error messages for common issues:**

```javascript
try {
  // ... upload logic
} catch (error) {
  console.error('Drive upload error:', error);
  
  // User-friendly error messages
  let userMessage = 'Failed to upload document to Google Drive.';
  
  if (error.message.includes('401') || error.message.includes('Invalid Credentials')) {
    userMessage = 'Your Google login has expired. Please log out and log back in.';
  } else if (error.message.includes('403') || error.message.includes('insufficient permission')) {
    userMessage = 'Permission denied. Make sure Google Drive access is enabled.';
  } else if (error.message.includes('404')) {
    userMessage = 'Drive folder not found. The folder may have been deleted.';
  }
  
  res.status(500).json({ 
    error: userMessage,
    details: error.message 
  });
}
```
