### STEP 5: Update Stage Progress When Tasks/Documents Complete
**Add this logic:**

```javascript
// When a task is marked complete
async function updateTaskStatus(taskId, isComplete) {
  await db.query(`
    UPDATE deal_tasks 
    SET status = ? 
    WHERE id = ?
  `, [isComplete ? 'completed' : 'pending', taskId]);
  
  // Check if all tasks in this stage are complete
  await checkStageCompletion(taskId);
}

// When a document is marked complete
async function updateDocumentStatus(docId, isComplete) {
  await db.query(`
    UPDATE deal_documents 
    SET status = ? 
    WHERE id = ?
  `, [isComplete ? 'completed' : 'pending', docId]);
  
  // Check if all documents in this stage are complete
  await checkStageCompletion(docId);
}

// Check if stage should be marked complete
async function checkStageCompletion(taskOrDocId) {
  // Get the stage for this task/document
  const stage = await db.query(`
    SELECT ds.* 
    FROM deal_stages ds
    JOIN deal_tasks dt ON dt.deal_stage_id = ds.id
    WHERE dt.id = ?
  `, [taskOrDocId]);
  
  if (!stage) return;
  
  // Check if all required tasks are complete
  const incompleteTasks = await db.query(`
    SELECT COUNT(*) as count
    FROM deal_tasks
    WHERE deal_stage_id = ?
    AND is_required = true
    AND status != 'completed'
  `, [stage.id]);
  
  // Check if all required documents are complete
  const incompleteDocs = await db.query(`
    SELECT COUNT(*) as count
    FROM deal_documents
    WHERE stage_id = ?
    AND is_required = true
    AND status != 'completed'
  `, [stage.id]);
  
  // If all required items are complete, mark stage as complete
  if (incompleteTasks.count === 0 && incompleteDocs.count === 0) {
    await db.query(`
      UPDATE deal_stages
      SET status = 'completed'
      WHERE id = ?
    `, [stage.id]);
    
    // Move to next stage
    const nextStage = await db.query(`
      SELECT * FROM deal_stages
      WHERE deal_id = ?
      AND stage_order > ?
      ORDER BY stage_order ASC
      LIMIT 1
    `, [stage.deal_id, stage.stage_order]);
    
    if (nextStage) {
      await db.query(`
        UPDATE deal_stages
        SET status = 'in_progress'
        WHERE id = ?
      `, [nextStage.id]);
    }
  }
}
```
