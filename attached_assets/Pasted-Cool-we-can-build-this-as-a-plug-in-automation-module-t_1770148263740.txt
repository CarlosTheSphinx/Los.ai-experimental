# A) Data model in Drizzle (Postgres)

### 1) `automations` table

Stores user-created automations.

```ts
// server/db/schema/automations.ts
import { pgTable, text, boolean, timestamp, jsonb, uuid } from "drizzle-orm/pg-core";

export const automations = pgTable("automations", {
  id: uuid("id").primaryKey().defaultRandom(),
  name: text("name").notNull(),
  isEnabled: boolean("is_enabled").notNull().default(true),

  // Examples: "daily_digest", "deal_stage_changed", "doc_uploaded", "inactivity"
  triggerType: text("trigger_type").notNull(),
  triggerConfig: jsonb("trigger_config").notNull().default({}),  // e.g. { hourLocal: 9 }

  // array of rules: [{ field, op, value }]
  conditions: jsonb("conditions").notNull().default([]),

  // array of actions: [{ type, params }]
  actions: jsonb("actions").notNull().default([]),

  // optional: guardrails to prevent spam
  throttle: jsonb("throttle").notNull().default({}),             // e.g. { maxPerDay: 1 }

  createdBy: text("created_by").notNull(),
  createdAt: timestamp("created_at", { withTimezone: true }).notNull().defaultNow(),
  updatedAt: timestamp("updated_at", { withTimezone: true }).notNull().defaultNow(),
});
```

# D) Job runner with 15-minute cron + per-day schedule

Since you said polling is fine, we’ll keep backend scheduling simple too.

### Approach

* Run a **cron** every 15 minutes (or every hour) on Replit Deployments.
* The cron executes:

  1. load enabled `daily_digest` automations
  2. if it’s the right hour (e.g. 9am user timezone) and not already sent today:

     * generate digests
     * send

### Locking

Use `automation_locks` so it only runs once per interval even if multiple workers.

---

## Core daily digest job code (skeleton)

```ts
// server/jobs/dailyDigest.ts
import { and, eq, gt, sql } from "drizzle-orm";
import { db } from "../db/client";
import { automations, automationLocks, loanDigestState, loanUpdates } from "../db/schema";
import { sendEmail } from "../services/email";
import { createOrGetThread, postMessage } from "../services/messages"; // your module

function startOfTodayUTC() {
  const d = new Date();
  d.setUTCHours(0,0,0,0);
  return d;
}

async function acquireLock(key: string, ttlSeconds: number) {
  const lockedUntil = new Date(Date.now() + ttlSeconds * 1000);
  // Use an UPSERT lock
  const result = await db.execute(sql`
    INSERT INTO automation_locks (key, locked_until)
    VALUES (${key}, ${lockedUntil})
    ON CONFLICT (key)
    DO UPDATE SET locked_until = EXCLUDED.locked_until
    WHERE automation_locks.locked_until < NOW()
    RETURNING key;
  `);
  return (result.rows?.length ?? 0) > 0;
}

export async function runDailyDigests() {
  const gotLock = await acquireLock("daily_digest_runner", 60); // 1 minute lock
  if (!gotLock) return;

  const autos = await db.select().from(automations).where(eq(automations.triggerType, "daily_digest"));
  const now = new Date();

  for (const a of autos) {
    if (!a.isEnabled) continue;

    const hourLocal = (a.triggerConfig as any)?.hourLocal ?? 9; // default 9am local
    // Simple: run at approx that hour UTC for now
    // Better: store user timezone and compute per user; can be v2.
    if (now.getUTCHours() !== hourLocal) continue;

    await runDigestForAutomation(a.id);
  }
}

async function runDigestForAutomation(automationId: string) {
  // Find loans/users that should receive digest.
  // You’ll implement this based on your schema (loans in active pipeline).
  const targets = await getActiveLoanTargets(); // [{ loanId, userId, userEmail, dealId? }]

  for (const t of targets) {
    const shouldSend = await shouldSendToday(t.loanId, t.userId);
    if (!shouldSend) continue;

    const outstandingDocs = await getOutstandingDocs(t.loanId);
    const updates = await getUpdatesSinceLastDigest(t.loanId, t.userId);

    // If nothing to say, you can skip (or still send a “no updates” message if you want).
    if (outstandingDocs.length === 0 && updates.length === 0) {
      await markDigestSent(t.loanId, t.userId);
      continue;
    }

    const portalLink = `${process.env.APP_BASE_URL}/app/loans/${t.loanId}/documents`;

    const body = renderDigestBody({ outstandingDocs, updates, portalLink });

    // In-app: post as notification (or message)
    const thread = await createOrGetThread({ userId: t.userId, dealId: t.dealId ?? null, subject: `Loan Updates` });
    await postMessage({
      threadId: thread.id,
      senderRole: "system",
      type: "notification",
      body
    });

    // Email
    await sendEmail({
      to: t.userEmail,
      subject: `Daily Loan Update: ${t.loanId}`,
      html: renderDigestEmailHtml({ outstandingDocs, updates, portalLink })
    });

    await markDigestSent(t.loanId, t.userId);
  }
}

async function shouldSendToday(loanId: string, userId: string) {
  const state = await db.select().from(loanDigestState).where(eq(loanDigestState.loanId, loanId));
  const row = state[0];

  // If no state exists, send
  if (!row) return true;

  const last = new Date(row.lastDigestAt);
  const today = startOfTodayUTC();
  return last < today;
}

async function markDigestSent(loanId: string, userId: string) {
  await db.execute(sql`
    INSERT INTO loan_digest_state (loan_id, user_id, last_digest_at)
    VALUES (${loanId}, ${userId}, NOW())
    ON CONFLICT (loan_id)
    DO UPDATE SET last_digest_at = NOW();
  `);
}

async function getUpdatesSinceLastDigest(loanId: string, userId: string) {
  const state = await db.select().from(loanDigestState).where(eq(loanDigestState.loanId, loanId));
  const since = state[0]?.lastDigestAt ?? new Date(0);

  const rows = await db
    .select()
    .from(loanUpdates)
    .where(and(eq(loanUpdates.loanId, loanId), eq(loanUpdates.userId, userId), gt(loanUpdates.createdAt, since)));

  return rows.map(r => r.summary);
}

function renderDigestBody({ outstandingDocs, updates, portalLink }: any) {
  const lines: string[] = [];
  if (updates.length) {
    lines.push("✅ Updates since last check:");
    for (const u of updates) lines.push(`• ${u}`);
    lines.push("");
  }
  if (outstandingDocs.length) {
    lines.push("📄 Outstanding documents needed:");
    for (const d of outstandingDocs) lines.push(`• ${d.name}${d.note ? ` (${d.note})` : ""}`);
    lines.push("");
    lines.push(`Upload here: ${portalLink}`);
  } else {
    lines.push(`No outstanding documents right now. Portal: ${portalLink}`);
  }
  return lines.join("\n");
}

function renderDigestEmailHtml({ outstandingDocs, updates, portalLink }: any) {
  const u = updates.map((x: string) => `<li>${escapeHtml(x)}</li>`).join("");
  const d = outstandingDocs.map((x: any) => `<li>${escapeHtml(x.name)}${x.note ? ` (${escapeHtml(x.note)})` : ""}</li>`).join("");

  return `
    <div style="font-family: Arial, sans-serif; line-height:1.5">
      ${updates.length ? `<h3>Updates since last check</h3><ul>${u}</ul>` : ""}
      <h3>Outstanding documents</h3>
      ${outstandingDocs.length ? `<ul>${d}</ul>` : "<p>None 🎉</p>"}
      <p><a href="${portalLink}">Upload documents / view loan</a></p>
    </div>
  `;
}

function escapeHtml(s: string) {
  return s.replace(/&/g,"&amp;").replace(/</g,"&lt;").replace(/>/g,"&gt;").replace(/"/g,"&quot;").replace(/'/g,"&#039;");
}
```

# E) How admin actions become “tomorrow’s digest”

You said:

> “Whenever a step happens on the lender/admin end that update is reflected on the summary sent the next day.”

That’s exactly what `loan_updates` is for.

### Add one helper function and use it anywhere relevant

```ts
// server/services/loanUpdates.ts
import { db } from "../db/client";
import { loanUpdates } from "../db/schema/loanUpdates";

export async function recordLoanUpdate({
  loanId,
  userId,
  type,
  summary,
  meta = {}
}: {
  loanId: string;
  userId: string;
  type: string;
  summary: string;
  meta?: any;
}) {
  await db.insert(loanUpdates).values({
    loanId,
    userId,
    type,
    summary,
    meta
  });
}
```

Then anywhere in admin workflow you already have:

* stage changed
* doc requested
* underwriting conditions added
* appraisal ordered
* clear to close
* etc.

You add **one line**:

```ts
await recordLoanUpdate({
  loanId,
  userId: borrowerOrBrokerUserId,
  type: "stage_change",
  summary: `Loan moved to Underwriting.`,
  meta: { from: "Processing", to: "Underwriting" }
});
```

This does not change existing logic. It just writes a ledger entry.
